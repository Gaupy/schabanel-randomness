\newcommand{\Algotitle}[1]{
\addtocounter{prog}{1}
\put(2,-3.5){\textbf{Algorithm \theprog.} #1}
\linethickness{0.8pt}
\put(0,\lin){\line(1,0){130}}
\linethickness{0.3pt}
\advance \lin by -5
\put(0,\lin){\line(1,0){130}}

}

\newcommand{\algoline}[1]{
\advance \lin by -4
\addtocounter{nbr}{1}
\ifnum \thenbr<10 \vari=4 \else \vari=2 \fi
\put(\vari,\lin){\thenbr}\put(10,\lin){#1}
}

\linethickness{0.3pt}
\advance \lin by -4
\put(0,\lin){\line(1,0){130}}
}
